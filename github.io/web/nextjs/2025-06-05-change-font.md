기본적으로 next 패키지에 많은 폰트가 들어가 있고 이를 사용해 변경

#### Noto Sans 폰트 적용

> app/layout.tsx 변경

```tsx
// app/layout.js (App Router)
import "./globals.css";
import { Noto_Sans_KR } from "next/font/google"; // 폰트 모듈 임포트

// 폰트 객체 생성
const notoSansKr = Noto_Sans_KR({
  weight: ["400", "700"], // 사용할 폰트 두께 지정
  subsets: ["latin"], // 필요한 서브셋 지정 (보통 'latin' 또는 'korean')
  display: "swap",
});

export const metadata = {
  title: "My App",
  description: "Generated by create next app",
};

export default function RootLayout({ children }) {
  return (
    <html lang="ko" className={notoSansKr.className}>
      {" "}
      {/* 폰트 클래스 적용 */}
      <body>{children}</body>
    </html>
  );
}
```
